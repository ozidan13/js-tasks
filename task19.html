الهدف التعليمي

تعلم كيفية تنفيذ العد التنازلي للـ Event باستخدام JavaScript، مع إضافة بعض التأثيرات المتقدمة لتحسين تجربة المستخدم.

JavaScript
// Countdown Example
const eventDate = new Date("Feb 28, 2025 15:00:00").getTime();

let countdownInterval = setInterval(function () {
    const now = new Date().getTime();
    const distance = eventDate - now;

    const days = Math.floor(distance / (1000 * 60 * 60 * 24));
    const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
    const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
    const seconds = Math.floor((distance % (1000 * 60)) / 1000);

    document.getElementById("countdown-display").innerHTML =
        `
            
العد التنازلي للحدث:

            
    
                
  • أيام: ${days}
    • 
                
  • ساعات: ${hours}
    • 
                
  • دقائق: ${minutes}
    • 
                
  • ثواني: ${seconds}
    • 
            

        `;

    if (distance < 0) {
        clearInterval(countdownInterval);
        document.getElementById("countdown-display").innerHTML = "الحدث قد بدأ!";
    }
}, 1000);

// Start Countdown button functionality
document.getElementById("start-countdown").addEventListener("click", function() {
    countdownInterval; // Starts countdown when the button is clicked.
});
