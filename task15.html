// Data Cleaning Example
const names = ["Mohamed", "Sara123", "Ali@!", "Ahmed"];
names.push("Osama"); // Adding Osama

// Cleaning and Filtering
const cleanedNames = [...names.map(name => name.trim().replace(/[^a-zA-Zء-ي ]/g, ""))];
const updatedNames = cleanedNames.filter(name => name !== "Sara");

// Displaying Results
console.log(updatedNames);
