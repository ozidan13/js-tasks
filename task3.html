الهدف التعليمي

تعلم كيفية التحقق من أيام الأسبوع باستخدام JavaScript مع تطبيق مبادئ التحقق من المدخلات وعرض النتائج بشكل تفاعلي.

ما هو يومك المفضل؟

الشرح البرمجي 
document.getElementById("dayForm").addEventListener("submit", function(event) {
    event.preventDefault();
    const dayInput = document.getElementById("dayInput").value.trim().toLowerCase();
    const messageDiv = document.getElementById("message");
    let message;

    switch (dayInput) {
        case 'الجمعة':
            message = "الجمعة هو يومك المفضل! استمتع بعطلة نهاية الأسبوع!";
            break;
        case 'السبت':
            message = "السبت يوم رائع للاسترخاء!";
            break;
        // ... بقية الحالات ...
        default:
            message = "يوم غير صالح. الرجاء إدخال يوم صحيح من أيام الأسبوع!";
            messageDiv.className = "message error";
    }

    messageDiv.textContent = message;
    if (!messageDiv.className.includes("error")) {
        messageDiv.className = "message";
    }
});
